32  // Screen's width needs to fit 20 cells
33  // Screen's height needs to fit 11 cells
34  // 800 / 20 = 40 SCREEN_WIDTH / CELLS_X
35  // 600 / 11 = 54 SCREEN_HEIGHT / CELLS_Y
36  // Pick the smaller number to make cells square and ensure they fit in both X and Y.
37  // So 40x40 px would be the maximum cell size

48  // was filled with black pixels as well:
49  // http://i.imgur.com/FEP30q6.png
50  // So the game frame on X axis fills 20 cells of 3x3 pixels (20x3), each separated by
51  // a single pixel (20*3 + 19*1). The frame itself is 1 pixel wide on each side
52  // (20*3 + 19*1 + 2) and the frame is also separated from the cells by 1 pixel on each
53  // side (20*3 + 19*1 + 2 + 2) = 83
54  static constexpr int GRID_WIDTH = CELLS_X * CELL_SIZE + (CELLS_X-1) * CELL_LARGE_PIXEL_OFFSET + 2 * CELL_LARGE_PIXEL_OFFSET
