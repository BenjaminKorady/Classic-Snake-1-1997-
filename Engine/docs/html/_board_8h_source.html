33  //To figure out how big each cell is going to be:
34  // Screen's width needs to fit 20 cells
35  // Screen's height needs to fit 11 cells
36  // 800 / 20 = 40 SCREEN_WIDTH / CELLS_X
37  // 600 / 11 = 54 SCREEN_HEIGHT / CELLS_Y
38  // Pick the smaller number to make cells square and ensure they fit in both X and Y.
39  // So 40x40 px would be the maximum cell size
40  // In the classic Snake, each cell was 3x3 black pixels separated by 1 blank pixel:

49  // And if the snake segments were connected, the 1x3 line of pixels inbetween them
50  // was filled with black pixels as well:
51  // http://i.imgur.com/FEP30q6.png
52  // So the game frame on X axis fills 20 cells of 3x3 pixels (20x3), each separated by
53  // a single pixel (20*3 + 19*1). The frame itself is 1 pixel wide on each side
54  // (20*3 + 19*1 + 2) and the frame is also separated from the cells by 1 pixel on each
55  // side (20*3 + 19*1 + 2 + 2) = 83
56  static constexpr int GRID_WIDTH = CELLS_X * CELL_SIZE + (CELLS_X-1) * CELL_LARGE_PIXEL_OFFSET + 2 * CELL_LARGE_PIXEL_OFFSET
57  + 2 * CELL_LARGE_PIXEL_OFFSET; //83

60  // (11*3 + 10*1 + 2 + 2) = 47
61  static constexpr int GRID_HEIGHT = CELLS_Y * CELL_SIZE + (CELLS_Y - 1) * CELL_LARGE_PIXEL_OFFSET + 2 * CELL_LARGE_PIXEL_OFFSET
62  + 2 * CELL_LARGE_PIXEL_OFFSET; //47

64  // As the PC screen resolution is much larger than the old phones' resolution. The actual PC
65  // pixels are smaller to the human eye and playing on a 83x47 resolution would be nothing special.
66  // So we create a sort of "large pixel". What seems like 1 pixel on the phone, will be more pixels
67  // on the PC monitor. The largest possible value for each pixel would be:
68  // (800 / 83) which is approximately 9.64 so to round it up: 1 phone pixel = 9 PC monitor pixels
